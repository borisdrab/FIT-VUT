33 /**
34 * @brief Konvertuje infixový zápis tokenov do postfixového zápisu (RPN - Reverse Polish Notation).
35 * @param infix Zoznam tokenov v infixovom zápise.
36 * @return Zoznam tokenov v postfixovom zápise.
37 * @exception InvalidOperationException Vyvolaná, ak sú zátvorky nesprávne alebo neznámy token.
38 */
39 private static List<string> ConvertInfixTokensToPostfix(List<string> infix)
40 {
41 var outputQueue = new List<string>();
42 var operatorStack = new Stack<string>();
43
44 var operatorData = Tokens.OperatorData;
45
46 foreach (var token in infix)
47 {
48 if (double.TryParse(token, NumberStyles.Float, CultureInfo.InvariantCulture, out _))
49 {
50 outputQueue.Add(token);
51 }
52 else if (operatorData.TryGetValue(token, out var currentInfo))
53 {
54 while (operatorStack.Count > 0 && operatorData.TryGetValue(operatorStack.Peek(), out var topInfo))
55 {
56 var shouldPop =
57 (currentInfo.rightAssociative && currentInfo.precedence < topInfo.precedence) ||
58 (!currentInfo.rightAssociative && currentInfo.precedence <= topInfo.precedence);
59
60 if (!shouldPop)
61 break;
62
63 outputQueue.Add(operatorStack.Pop());
64 }
65
66 operatorStack.Push(token);
67 }
68 else if (token == Tokens.LEFT_PARENTHESIS)
69 {
70 operatorStack.Push(token);
71 }
72 else if (token == Tokens.RIGHT_PARENTHESIS)
73 {
74 while (operatorStack.Count > 0 && operatorStack.Peek() != Tokens.LEFT_PARENTHESIS)
75 {
76 outputQueue.Add(operatorStack.Pop());
77 }
78
79 if (operatorStack.Count == 0 || operatorStack.Peek() != Tokens.LEFT_PARENTHESIS)
80 {
81 throw new InvalidOperationException("Mismatched parentheses");
82 }
83
84 operatorStack.Pop();
85
86 // Pop the function after ")"
87 if (operatorStack.Count > 0 && operatorData.ContainsKey(operatorStack.Peek()) && operatorStack.Peek() == Tokens.LOGARITHM)
88 {
89 outputQueue.Add(operatorStack.Pop());
90 }
91 }
92 else
93 {
94 throw new InvalidOperationException($"Unknown token: {token}");
95 }
96 }
97
98 while (operatorStack.Count > 0)
99 {
100 var op = operatorStack.Pop();
101 if (op is Tokens.LEFT_PARENTHESIS or Tokens.RIGHT_PARENTHESIS)
102 throw new InvalidOperationException("Mismatched parentheses in expression");
103
104 outputQueue.Add(op);
105 }
106
107 return outputQueue;
108 }

119 private static double EvaluatePostfixTokens(List<string> postfix)
120 {
121 var stack = new Stack<double>();
122
123 foreach (var token in postfix)
124 {
125 if (IsOperand(token, out var number))
126 {
127 stack.Push(number);
128 }
129 else if (Tokens.OperatorData.ContainsKey(token))
130 {
131 var b = stack.Pop();
132 var a = 0.0;
133 if (token is not Tokens.FACTORIAL and not Tokens.LOGARITHM)
134 {
135 if (stack.TryPeek(out _))
136 {
137 a = stack.Pop();
138 }
139 }
140
141 var result = token switch
142 {
143 Tokens.ADD => Operations.Add(a, b),
144 Tokens.SUBTRACT => Operations.Subtract(a, b),
145 Tokens.MULTIPLY => Operations.Multiply(a, b),
146 Tokens.DIVIDE => Operations.Divide(a, b),
147 Tokens.POWER => Operations.Power(a, b),
148 Tokens.FACTORIAL => Operations.Factorial((int)b),
149 Tokens.LOGARITHM => Operations.Log(b, 10),
150 _ => throw new InvalidOperationException($"Unknown operator: {token}")
151 };
152
153 stack.Push(result);
154 }
155 else
156 {
157 throw new InvalidOperationException($"Unknown token: {token}");
158 }
159 }
160
161 return stack.Pop();
162 }

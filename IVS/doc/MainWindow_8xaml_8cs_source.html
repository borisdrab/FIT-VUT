152 /// <summary>
153 /// Umoznuje zadavat vstup z klavesnice (cisla, operatory, zatvorky, Enter, ...).
154 /// </summary>
155 private void MainWindow_KeyDown(object sender, KeyEventArgs e)
156 {
157 if (e.Key >= Key.D0 && e.Key <= Key.D9)
158 {
159 currentInput += (e.Key - Key.D0).ToString();
160 }
161 else if (e.Key >= Key.NumPad0 && e.Key <= Key.NumPad9)
162 {
163 currentInput += (e.Key - Key.NumPad0).ToString();
164 }
165 else if (e.Key == Key.OemPeriod || e.Key == Key.Decimal)
166 {
167 Decimal_Click(this, new RoutedEventArgs());
168 return;
169 }
170 else if (e.Key == Key.Add)
171 currentInput += "+";
172 else if (e.Key == Key.Subtract)
173 currentInput += "-";
174 else if (e.Key == Key.Multiply)
175 currentInput += "*";
176 else if (e.Key == Key.Divide)
177 currentInput += "/";
178 else if (e.Key == Key.Enter || e.Key == Key.Return)
179 {
180 e.Handled = true;
181 Equals_Click(this, new RoutedEventArgs());
182 return;
183 }
184 else if (e.Key == Key.D9 && Keyboard.Modifiers == ModifierKeys.Shift)
185 currentInput += "(";
186 else if (e.Key == Key.D0 && Keyboard.Modifiers == ModifierKeys.Shift)
187 currentInput += ")";
188
189 Display.Text = currentInput;
190 }
191 }

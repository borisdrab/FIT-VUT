32 /// <summary>
33 /// Vypočíta smerodajnú odchýlku zoznamu hodnôt.
34 /// </summary>
35 /// @brief Vypočíta smerodajnú odchýlku zoznamu hodnôt.
36 /// @param values Zoznam čísel, pre ktoré sa vypočíta smerodajná odchýlka.
37 /// @return Smerodajná odchýlka vstupného zoznamu.
38 /// @exception ArgumentException Vyvolaná, ak je vstupný zoznam prázdny.
39 public static double StandardDeviation(List<double> values)
40 {
41 if (values.Count == 0)
42 throw new ArgumentException("List cannot be empty");
43
44 // Spočítame priemer ručne pomocou vlastných funkcií
45 double sum = 0;
46 foreach (var value in values)
47 {
48 sum = Add(sum, value);
49 }
50 double mean = Divide(sum, values.Count);
51
52 // Spočítame súčet štvorcov rozdielov
53 double sumOfSquares = 0;
54 foreach (var value in values)
55 {
56 double diff = Subtract(value, mean);
57 double square = Power(diff, 2);
58 sumOfSquares = Add(sumOfSquares, square);
59 }
60
61 double variance = Divide(sumOfSquares, values.Count);
62 return Root(variance, 2); // Druhá odmocnina pomocou tvojej Root funkcie
63 }

65 /// <summary>
66 /// Vypočíta faktoriál čísla.
67 /// </summary>
68 /// @brief Vypočíta faktoriál čísla.
69 /// @param number Číslo, ktorého faktoriál sa má vypočítať.
70 /// @return Faktoriál čísla.
71 /// @exception ArgumentException Vyvolaná, ak je číslo záporné.
72 public static double Factorial(int number)
73 {
74 // Kontrola pre záporné čísla
75 if (number < 0)
76 throw new ArgumentException("Factorial is not defined for negative numbers.");
77
78 // Ak je už v cache, vrátime uložený výsledok
79 if (FactorialCache.ContainsKey(number))
80 return FactorialCache[number];
81
82 // Základný prípad: 0! = 1
83 if (number == 0)
84 {
85 FactorialCache[0] = 1; // Uložíme 0! do cache
86 return 1;
87 }
88
89 // Skontrolujeme, či máme v cache predchádzajúci faktorál (n-1)!
90 var prevFact = Factorial(number - 1);
91
92 // Vypočítame aktuálny faktorál
93 var fact = number * prevFact;
94
95 // Uložíme aktuálny faktorál do cache
96 FactorialCache[number] = fact;
97
98 return fact;
99 }

101 /// <summary>
102 /// Vypočíta logaritmus čísla pri zadanom základe.
103 /// </summary>
104 /// @brief Vypočíta logaritmus čísla pri zadanom základe.
105 /// @param a Číslo, z ktorého sa počíta logaritmus (musí byť kladné).
106 /// @param base Základ logaritmu (musí byť kladný a rôzny od 1).
107 /// @return Výsledok logaritmu čísla a pri základe base.
108 /// @exception IndexOutOfRangeException Vyvolaná, ak a alebo base nie sú platné hodnoty.
109 public static double Log(double a, double @base)
110 {
111 if (a <= 0)
112 throw new IndexOutOfRangeException("Logarithm argument must be positive");
113
114 if (@base <= 0 || Math.Abs(@base - 1) < double.Epsilon)
115 throw new IndexOutOfRangeException("Logarithm base must be positive and not equal to 1");
116
117 return Math.Log(a, @base);
118 }

121 /// <summary>
122 /// Vypočíta n-tú odmocninu z čísla.
123 /// </summary>
124 /// @brief Vypočíta n-tú odmocninu z čísla.
125 /// @param a Číslo, z ktorého sa počíta odmocnina.
126 /// @param exponent Exponent (stupeň odmocniny, musí byť nenulový).
127 /// @return Výsledok n-tej odmocniny čísla a.
128 /// @exception IndexOutOfRangeException Vyvolaná, ak exponent je nulový alebo je odmocňované číslo záporné a exponent párny.
129 public static double Root(double a, double exponent)
130 {
131 if (exponent == 0)
132 throw new IndexOutOfRangeException("Exponent must be non-zero");
133
134 // Ak exponent je párne číslo, záporný základ nie je povolený
135 if (a < 0 && exponent % 2 == 0)
136 throw new IndexOutOfRangeException("Cannot calculate even root of negative number");
137
138 // Ak exponent nie je celé číslo a základ je záporný, vyhodíme výnimku (komplexné čísla)
139 if (a < 0 && exponent % 1 != 0)
140 throw new IndexOutOfRangeException("Cannot calculate root of negative number with non-integer exponent");
141
142 // Ak základ je záporný a exponent je nepárny, môžeme spočítať koreň
143 if (a < 0)
144 return -Math.Pow(-a, 1.0 / exponent);
145
146 // Pre kladné čísla môžeme použiť klasický výpočet
147 return Math.Pow(a, 1.0 / exponent);
148 }

150 /// <summary>
151 /// Vypočíta umocnenie čísla na daný exponent.
152 /// </summary>
153 /// @brief Vypočíta umocnenie čísla na daný exponent.
154 /// @param num Základ mocniny.
155 /// @param exp Exponent.
156 /// @return Výsledok umocnenia num na exp.
157 /// @exception ArgumentException Vyvolaná, ak je základ aj exponent nula alebo ak je základ záporný a exponent nie je celé číslo.
158 public static double Power(double num, double exp)
159 {
160 // Špeciálny prípad, kedy je základ aj exponent nula, čo je matematicky nedefinované.
161 if (num == 0 && exp == 0)
162 throw new ArgumentException("0^0 is undefined");
163
164 // Kontrola pre záporný základ a neceločíselný exponent. Výsledok by bol komplexné číslo, ktoré táto metóda neobsluhuje.
165 if (num < 0 && exp % 1 != 0)
166 throw new ArgumentException("Negative base with non-integer exponent is undefined");
167
168 // Použitie vstavané funkcie Math.Pow na výpočet mocniny.
169 return Math.Pow(num, exp);
170 }

172 /// <summary>
173 /// Sčíta dve čísla.
174 /// </summary>
175 /// @brief Sčíta dve čísla.
176 /// @param a Prvé číslo.
177 /// @param b Druhé číslo.
178 /// @return Súčet čísiel a a b.
179 /// @exception OverflowException Vyvolaná, ak pretečie rozsah pre kladné čísla.
180 public static double Add(double a, double b)
181 {
182 // Kontrola pretečenia pri sčítaní dvoch kladných čísel. Ak je výsledok záporný, došlo k pretečeniu rozsahu double.
183 if (a > 0 && b > 0 && a + b < 0)
184 throw new OverflowException("Arithmetic operation resulted in an overflow.");
185
186 // Vrátenie súčtu dvoch čísel.
187 return a + b;
188 }

190 /// <summary>
191 /// Odčíta druhé číslo od prvého.
192 /// </summary>
193 /// @brief Odčíta druhé číslo od prvého.
194 /// @param a Prvé číslo.
195 /// @param b Druhé číslo.
196 /// @return Rozdiel čísiel a a b.
197 /// @exception OverflowException Vyvolaná, ak pretečie rozsah pre záporné čísla.
198 public static double Subtract(double a, double b)
199 {
200 // Kontrola pretečenia pri odčítaní dvoch záporných čísel. Ak je výsledok kladný, došlo k pretečeniu rozsahu double.
201 if (a < 0 && b < 0 && a - b > 0)
202 throw new OverflowException("Arithmetic operation resulted in an overflow.");
203
204 // Vrátenie rozdielu dvoch čísel.
205 return a - b;

208 /// <summary>
209 /// Vynásobí dve čísla.
210 /// </summary>
211 /// @brief Vynásobí dve čísla.
212 /// @param a Prvé číslo.
213 /// @param b Druhé číslo.
214 /// @return Súčin čísiel a a b.
215 /// @exception OverflowException Vyvolaná, ak pretečie rozsah pre výsledok násobenia.
216 public static double Multiply(double a, double b)
217 {
218 // Vypočítanie výsledku násobenia.
219 var res = a * b;
220 // Kontrola pretečenia. Ak majú pôvodné čísla rovnaké znamienko a výsledok je záporný, alebo ak majú opačné znamienka a výsledok je kladný, došlo k pretečeniu.
221 if ((((a > 0 && b > 0) || (a < 0 && b < 0)) && res < 0) ||
222 (((a > 0 && b < 0) || (a < 0 && b > 0)) && res > 0))
223 throw new OverflowException("Arithmetic operation resulted in an overflow.");
224
225 // Vrátenie výsledku násobenia.
226 return res;
227 }
